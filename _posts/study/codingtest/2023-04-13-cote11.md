### 🚀 나의 풀이 ⭕

```python

import sys 


if __name__=="__main__":
    # input = sys.stdin.readline
    arr = list(map(int, input().split()))
    up=0
    down=0 
    for i in range(1,len(arr)):
        if arr[i]>arr[i-1]:
            up+=1
            down=0
        else:
            up=0
            down+=1
    if up !=7 and down !=7:
        print('mixed')
    else:
        if up==7:
            print('ascending')
        if down==7:
            print('descending')

```
