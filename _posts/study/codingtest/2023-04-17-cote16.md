from collections import deque

def BFS(n, computers, com, visited):
    visited[com]=True
    Q=deque()
    Q.append(com)
    while Q:
        com=Q.popleft()
        visited[com]=True 
        for connect in range(n):
            if connect != com and computers[com][connect]==1:
